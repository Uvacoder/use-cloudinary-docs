---
title: useImage
---


The `useImage` hook gives you the ability to call an image from your Cloudinary account and apply transformations. The result is a URL generated by the SDK. 

```jsx
import { useImage } from 'use-cloudinary';

function Image({ cloudName, publicId, transforms }) {
  const { getImage, data, status, error } = useImage({ cloud_name: cloudName });

  React.useEffect(() => {
    getImage({
      public_id: publicId,
      transform_options: {
        ...transforms
      }
    })
  }, [])

  if (status === "loading") return <p>Loading...</p>;
  if (status === "error") return <p>{error.message}</p>;

  return <img src={data} alt="Transformed from Cloudinary" />
}
```
